This class takes care of polyfilling some modules.

### Node Blob

```ts
import { Polyfill } from '@aracna/core'

// will log Blob
Polyfill.blob().then(() => console.log(Blob))
```

### Node Fetch

```ts
import { Polyfill } from '@aracna/core'

Polyfill.fetch().then(() => {
  // will log fetch, Headers, Request and Response
  console.log(fetch, Headers, Request, Response)
})
```

### Node File

```ts
import { Polyfill } from '@aracna/core'

// will log File
Polyfill.file().then(() => console.log(File))
```

### Node FormData

```ts
import { Polyfill } from '@aracna/core'

// will log FormData
Polyfill.formData().then(() => console.log(FormData))
```
