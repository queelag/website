This is a base64 codec which leverages the rfc4648 module.

### Encode

```ts
import { Base64 } from '@aracna/core'

const encoded: string = Base64.encode('hello')

// will log "aGVsbG8="
console.log(encoded)
```

### Decode

```ts
import { Base64 } from '@aracna/core'

const decoded: string = Base64.decode('aGVsbG8=')

// will log "hello"
console.log(decoded)
```
