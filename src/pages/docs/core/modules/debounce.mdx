This class will run the target function only if the target function hasn't be called for the specified amount of time.

### Usage

```ts
import { Debounce } from '@queelag/core'

const search_debounce_name: string = 'search'
const search_debounce_function: Function = () => console.log('searching')

document.addEventListener('mousemove', () => {
  // if you don't move your mouse for 500ms and then you move it again the target function will be executed
  Debounce.handle(search_debounce_name, search_debounce_function, 500)
})
```
