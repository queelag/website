This class contains utilities for [Promise](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Promise).

### Synchronous Chain

```ts
import { PromiseUtils } from '@queelag/core'

// will always log 0 before and 1 after
PromiseUtils.chain(
  async () => console.log(0),
  async () => console.log(1)
)
```

### Synchronous Truthy Chain

```ts
import { PromiseUtils, rc } from '@queelag/core'

// will log 0 and 1 but will not log 2 since the previous promise returned a falsy value
PromiseUtils.truthyChain(
  async () => rc(() => console.log(0), true),
  async () => console.log(1),
  async () => console.log(2)
)

// will log 0 but will not log 1 since the previous promise returned an Error
PromiseUtils.truthyChain(
  async () => rc(() => console.log(0), true),
  async () => new Error(),
  async () => console.log(1)
)
```
