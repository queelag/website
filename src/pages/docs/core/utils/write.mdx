This class contains utilities for the Write paradygm.

### Find Mode

```ts
import { WriteUtils } from '@queelag/core'

// will log CREATE
console.log(WriteUtils.findMode({ id: '' }))

// will log UPDATE
console.log(WriteUtils.findMode({ id: '$id' }))
```

### Check Mode

```ts
import { WriteUtils } from '@queelag/core'

// will log true
console.log(WriteUtils.isModeCreate({ id: '' }))

// will log false
console.log(WriteUtils.isModeCreate({ id: '$id' }))

// will log true
console.log(WriteUtils.isModeUpdate({ id: '$id' }))

// will log false
console.log(WriteUtils.isModeUpdate({ id: '' }))
```
